function observe(obj) {
  if (typeof obj !== "object" || obj === null) {
    return;
  }
  new Observer(obj);
}

class Observer {
  constructor(value) {
    this.value = value;
    this.walk(value);
  }
  walk(obj) {
    Object.keys(obj).forEach((key) => {
      defineReactive(obj, key, obj[key]);
    });

}

class Compile {
  constructor(el, vm) {
    this.$vm = vm;
    this.$el = document.querySelector(el); // иҺ·еҸ–dom
    if (this.$el) {
      this.compile(this.$el);
    }
  }
  compile(el) {
    const childNodes = el.childNodes;
    Array.from(childNodes).forEach((node) => { // йҒҚеҺҶеӯҗе…ғзҙ 
      if (this.isElement(node)) { // еҲӨж–ӯжҳҜеҗҰдёәиҠӮзӮ№
        console.log("зј–иҜ‘е…ғзҙ " + node.nodeName);
      } else if (this.isInterpolation(node)) {
        console.log("зј–иҜ‘жҸ’еҖјж–Үжң¬" + node.textContent); // еҲӨж–ӯжҳҜеҗҰдёәжҸ’еҖјж–Үжң¬ 
        {{}}
      }
      if (node.childNodes && node.childNodes.length > 0) { // еҲӨж–ӯжҳҜеҗҰжңүеӯҗе…ғзҙ 
        this.compile(node); // еҜ№еӯҗе…ғзҙ иҝӣиЎҢйҖ’еҪ’йҒҚеҺҶ
      }
    });
  }
  isElement(node) {
    return node.nodeType === 1;
  }
  isInterpolation(node) {
    return node.nodeType === 3 && /\{\{(.*)\}\}/.test(node.textContent);
  }
}

function defineReactive(obj, key, val) {
  this.observe(val);
  const dep = new Dep();
  Object.defineProperty(obj, key, {
    get() {
      Dep.target && dep.addDep(Dep.target);// Dep.targetе°ұжҳҜWatcherзҡ„е®һдҫӢ
      return val;
    },
    set(newVal) {
      if (newVal === val) return;
      dep.notify(); // йҖҡзҹҘdepжү§иЎҢжӣҙж–°ж–№жі•
    },
  });
}

  }

class Vue {
  constructor(options) {
    this.$data = options.data

    //и°ғз”Ёж•°жҚ®еҠ«жҢҒзҡ„ж–№жі•
    observe(this.$data);

    //еұһжҖ§д»ЈзҗҶ
    Object.keys(this.$data).forEach((key)=>{
      Object.defineProperty(this, key,{
        enumerable:true,
        configurable:true,
        get() {
          return this.$data[key];
        },
        set(newValue) {
          this.$data[key] = newValue;
        }
      })
    })

    //и°ғз”ЁжЁЎжқҝзј–иҜ‘
    compile(options.el, this);
  }
}
function observe(obj){
  //дҪҝз”ЁйҖ’еҪ’з»ҷеӨҡеұӮеҜ№иұЎеұһжҖ§йғҪж·»еҠ getter,setterж–№жі•
  if (!obj || typeof obj!== 'object') {
    return
  }

  const dep = new Dep();

  //йҖҡиҝҮObject.keys(obj)иҺ·еҸ–еҲ°еҪ“еүҚobjзҡ„жҜҸдёӘеұһжҖ§
  Object.keys(obj).forEach((key)=>{
    let value = obj[key]
    //жҠҠvalueиҝҷдёӘеӯҗиҠӮзӮ№иҝӣиЎҢйҖ’еҪ’
    observe(value)
    Object.defineProperty(obj,key,{
      enumerable:true,
      configurable:true,
      get() {
        //д»Һwatcherжһ„йҖ еҮҪж•°йӮЈйҮҢи·іиҝҮжқҘжү§иЎҢиҝҷйҮҢпјҢеӣ дёәйӮЈйҮҢз»ҷvmеҸ–еҖјиөӢеҖјдјҡи§ҰеҸ‘иҝҷйҮҢзҡ„еҮҪж•°
        //Dep.target
        //еҸӘиҰҒеңЁиҝҷйҮҢжү§иЎҢдәҶиҝҷдёҖиЎҢпјҢйӮЈд№Ҳиў«newеҮәжқҘзҡ„watcherе®һдҫӢе°ұиў«ж”ҫеңЁdep.subsж•°з»„дёӯдәҶ
        Dep.target && dep.addSub(Dep.target);
        return value
      },
      set(newVal) {
        value = newVal
        observe(value);
        //йҖҡзҹҘжҜҸдёҖдёӘи®ўйҳ…иҖ…
        dep.notify();
      }
    })
  })
}

/**
 * жЁЎжқҝзј–иҜ‘
 * @param el жҳҜoption.el иҝҷйҮҢжҳҜ#app
 * @param vm vueе®һдҫӢпјҢе°ұжҳҜиҝҷдёӘthis
 */
function compile(el,vm){
  vm.$el = document.querySelector(el);    //жҠҠиҝҷдёӘDOMе…ғзҙ жҢӮиҪҪеҲ°vmе®һдҫӢдёҠ
  //еҲӣе»әж–ҮжЎЈзўҺзүҮ
  const fragment = document.createDocumentFragment();
  //жҠҠеҺҹDOMж ‘дёҠзҡ„жүҖжңүе…ғзҙ йғҪж”ҫиҝӣж–ҮжЎЈзўҺзүҮдёӯ
  while (childNode = vm.$el.firstChild){
    fragment.append(childNode);
  }
  //еңЁиҝҷйҮҢиҝӣиЎҢжЁЎжқҝзј–иҜ‘
  modelReplace(fragment);

  /**
   * е®ҡд№үжЁЎжқҝзј–иҜ‘еҮҪж•°
   * з”ұдәҺеҸҳйҮҸжҸҗеҚҮпјҢж–№жі•жҸҗеҚҮпјҢжүҖд»ҘеҚідҪҝе®ҡд№үеҲ°еӨ–еұӮпјҢи°ғз”Ёзҡ„ж—¶еҖҷд№ҹеҸҜд»ҘеҫҒзЁӢи°ғз”Ё
   * @param node
   */
  function modelReplace(node) {
    //е®ҡд№үдёҖдёӘжӯЈеҲҷиЎЁиҫҫејҸ,еҢ№й…ҚжҸ’еҖјиЎЁиҫҫејҸ
    const regMustache = /\{\{\s*(\S+)\s*\}\}/;
    //иҜҒжҳҺеҪ“еүҚзҡ„иҠӮзӮ№жҳҜж–Үжң¬еӯҗиҠӮзӮ№пјҢйңҖиҰҒиҝӣиЎҢжӯЈеҲҷжӣҝжҚў
    if (node.nodeType === 3){
      //ж–Үжң¬еӯҗиҠӮзӮ№жҳҜдёҖдёӘDOMеҜ№иұЎпјҢжғіиҰҒиҺ·еҸ–е…¶еҶ…е®№пјҢйңҖиҰҒи°ғз”ЁиҠӮзӮ№зҡ„textContentеұһжҖ§иҺ·еҸ–
      // console.log(node.textContent);
      const text = node.textContent;
      //иҝӣиЎҢеӯ—з¬ҰдёІзҡ„жӯЈеҲҷеҢ№й…Қ
      const execResult = regMustache.exec(text);
      // console.log(execResult)
      if (execResult){
        // console.log(vm[execResult[1]])
        //жҠҠжүҖжңүеұһжҖ§зҡ„еҖјйғҪе°ҶиҝӣиЎҢдёҖдёӘreduceйҒҚеҺҶ
        const value = execResult[1].split('.').reduce((newObj,key)=>{
          return newObj[key]}, vm);
        // console.log(value);
        //жҠҠйҖҡиҝҮreduceйҒҚеҺҶеҫ—еҲ°зҡ„еҖјиөӢеҖјз»ҷnode.textContent
        node.textContent = text.replace(regMustache,value);
        //еңЁиҝҷйҮҢеҲӣе»әWatcherе®һдҫӢ
        new Watcher(vm, execResult[1],(newValue)=>{
          node.textContent = text.replace(regMustache,newValue);
        })
      }
      //з»ҲжӯўйҖ’еҪ’жқЎд»¶
      return;
    }
    //еҲӨж–ӯеҪ“еүҚзҡ„nodeиҠӮзӮ№жҳҜеҗҰдёәINPUTиҫ“е…ҘжЎҶ
    if (node.nodeType === 1 && node.tagName.toUpperCase() === 'INPUT'){
      // const attrs = node.attributes;
      // console.log(attrs);
      //жҠҠдјӘж•°з»„иҪ¬жҚўжҲҗзңҹзҡ„ж•°з»„
      const attrs = Array.from(node.attributes);
      //жҹҘжүҫжҳҜеҗҰе…·еӨҮv-modelиҝҷдёӘеұһжҖ§
      const findResult = attrs.find((value) => value.name === 'v-model');
      // console.log(findResult)
      if (findResult){
        //иҺ·еҸ–еҲ°еҪ“еүҚv-modelзҡ„еұһжҖ§зҡ„еҖј
        const expStr = findResult.value
        // console.log(expStr);
        const value = expStr.split('.').reduce((newObj,key)=>newObj[key],vm);
        // console.log(res);
        node.value = value;
        //еҲӣе»әWatcherе®һдҫӢпјҢжҠҠжӣҙж–°зҡ„ж–№жі•ж”ҫеңЁйҮҢйқў
        new Watcher(vm, expStr,(newVal)=>{
          node.value = newVal;
        })

        //зӣ‘еҗ¬ж–Үжң¬жЎҶзҡ„иҫ“е…ҘдәӢд»¶пјҢжӢҝеҲ°ж–Үжң¬жЎҶжңҖж–°зҡ„еҖјжӣҙж–°еҲ°vmдёҠ
        node.addEventListener('input',(e)=>{
          // console.log(e.target.value)
          //йҖҡиҝҮжӢҝеҲ°еҖ’ж•°з¬¬дәҢйЎ№пјҢз„¶еҗҺдҪҝз”Ё[]зҡ„еҪўејҸиҝӣиЎҢиөӢеҖј
          //еҜ№ж•°з»„иҝӣиЎҢеҲҶеүІ
          const keyArr = expStr.split('.');
          //sliceжҲӘеҸ–ж•°з»„пјҢдёҚеҢ…жӢ¬еҸіеҢәй—ҙ
          const obj = keyArr.slice(0,keyArr.length-1).reduce((newObj,key)=>{return newObj[key]},vm);
          //иөӢеҖј
          obj[keyArr[keyArr.length-1]] = e.target.value;
        })
      }
    }

    //дёҚжҳҜж–Үжң¬иҠӮзӮ№,еҸҜиғҪжҳҜдёҖдёӘDOMе…ғзҙ пјҢйңҖиҰҒиҝӣиЎҢйҖ’еҪ’еӨ„зҗҶ
    node.childNodes.forEach((child)=>{modelReplace(child)});
  }

  //жҠҠж–ҮжЎЈзўҺзүҮзҡ„еҶ…е®№ж”ҫеҲ°ж–ҮжЎЈдёҠ
  vm.$el.appendChild(fragment);
}

class Dep {
  constructor() {
    //жүҖжңүзҡ„watcherйғҪиў«еӯҳеҲ°иҝҷдёӘж•°з»„дёӯ
    this.subs = [];
  }
  //еҫҖж•°з»„йҮҢйқўиҝҪеҠ watcher
  addSub(watcher){
    this.subs.push(watcher);
  }
  //йҖҡзҹҘ
  notify(){
    this.subs.forEach((watcher) => {
      watcher.update()
    })
  }
}

class Watcher {
  /**
   * @param vm   vmдёӯдҝқеӯҳзқҖжңҖж–°зҡ„ж•°жҚ®
   * @param key  еңЁvmдёҠдј—еӨҡзҡ„ж•°жҚ®дёӯпјҢж №жҚ®keyжқҘзЎ®е®ҡжӣҙж–°е“ӘдёҖдёӘеҖјпјҢеңЁnew watcherжңҹй—ҙпјҢжҢҮе®ҡwatcherеҜ№еә”зҡ„ж•°жҚ®зҡ„еҗҚеӯ—
   * @param cb   еҰӮдҪ•еҺ»жӣҙж–°иҮӘе·ұ
   */
  constructor(vm, key, cb) {
    this.vm = vm;
    this.key = key;
    this.cb = cb;
    //еңЁжһ„йҖ еҮҪж•°йҮҢйқўжҠҠwatcherеӯҳеҲ°ж•°з»„йҮҢйқў
    //дёӢйқўдёүиЎҢд»Јз Ғеҗ§еҲӣе»әзҡ„йўқwatcherе®һдҫӢеӯҳеҲ°Depе®һдҫӢзҡ„subsж•°з»„дёӯ
    Dep.target = this;      //thisе°ұжҳҜwatcherе®һдҫӢ
    //иҝҷдёҖиЎҢд»Јз Ғдё»иҰҒжҳҜи®©д»–еҺ»и§ҰеҸ‘getterж–№жі•
    key.split('.').reduce((newObj,k)=>newObj[k],vm);    //жҜҸдёҖж¬ЎеҲӣе»әе®һдҫӢпјҢдјҡеңЁиҝҷйҮҢеҸ–дёҖж¬ЎеҖјз»ҷvmпјҢдјҡи§ҰеҸ‘getter
    Dep.target = null;
  }
  //иғҪи®©еҸ‘еёғиҖ…йҖҡзҹҘжҲ‘
  update(){
    //жҠҠж–°зҡ„еҖјдј з»ҷеӣһи°ғеҮҪж•°,ж–°зҡ„еҖјжқҘиҮӘдәҺthis.key
    const value = this.key.split('.').reduce((newObj,key) => {
      return newObj[key]
    }, this.vm)
    this.cb(value);
  }
}
